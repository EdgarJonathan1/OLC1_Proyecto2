
EXP_NUM
	: MENOS EXP_NUM %prec UMENOS				
	| EXP_NUM MAS EXP_NUM			
	| EXP_NUM MENOS EXP_NUM		
	| EXP_NUM POR EXP_NUM			
	| EXP_NUM DIVIDIDO EXP_NUM	
	| PARIZQ EXP_NUM PARDER					
	| ENTERO											
	| DECIMAL											
	| IDENTIFICADOR										
;

EXP_REL
	: EXP_NUM MAYQUE EXP_NUM		
	| EXP_NUM MENQUE EXP_NUM		
	| EXP_NUM MAYIGQUE EXP_NUM	
	| EXP_NUM MENIGQUE EXP_NUM	
	| EXP_CAD DOBLEIG EXP_CAD			
	| EXP_CAD NOIG EXP_CAD			
;

EXP_LOG
	: EXP_REL AND EXP_REL     
	| EXP_REL OR EXP_REL 		
	| NOT EXP_REL							
	| EXP_REL								
;
